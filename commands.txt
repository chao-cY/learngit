Commonly used commands:

	git init     //cd to target folder

	git add  readme.txt  //new file "readme.txt"

	git add  license.txt   //new file "license.txt"

	git commit -m "add two files"   //commit to repo
 
	git status  //check the situation of files in this folder
	
	git log  //view versions before current one

	git log --pretty=oneline //simplify information of version

	git reflog //view all of the versions

	git reset --hard 2ewrrrd  // jump to version  2ewrrrd

	git diff HEAD commands.txt //show the difference between HEAD and workspace

	git checkout -- readme.txt //undo the modification

	git reset HEAD readme.txt //undo latest add
	
	git rm readme.txt // remove file from git

	git push origin main // up-to-date to remote retro

	git remote rm origin //delete the link between local and remote

	git remote add origin git@server-name:path/repo-name.git
	(git remote add origin https://github.com/chao-cY/learngit.git)
	(origin is a name that we give remote repo)

	git push -u origin master  //first time push

	git push origin master  //push
